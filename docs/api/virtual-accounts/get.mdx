# Get Virtual Account

Retrieve detailed information about a specific virtual account, including its status and deposit instructions.

## Method Signature

```typescript
align.virtualAccounts.get(accountId: string): Promise<VirtualAccount>
```

## Parameters

<ParamField path="accountId" type="string" required>
  The unique virtual account identifier
</ParamField>

## Returns

<ResponseField name="id" type="string">
  Unique identifier for the virtual account
</ResponseField>

<ResponseField name="customer_id" type="string">
  Associated customer identifier
</ResponseField>

<ResponseField name="status" type="string">
  Account status: `active`, `pending`, `inactive`
</ResponseField>

<ResponseField name="source_currency" type="string">
  Fiat currency for deposits
</ResponseField>

<ResponseField name="source_rails" type="string">
  Payment rails type
</ResponseField>

<ResponseField name="destination_token" type="string">
  Cryptocurrency token
</ResponseField>

<ResponseField name="destination_network" type="string">
  Blockchain network
</ResponseField>

<ResponseField name="deposit_instructions" type="object">
  Bank account details for deposits
</ResponseField>

## Examples

<Tabs>
  <Tab title="TypeScript">
    ```typescript
    import Align from "@tolbel/align";

    const align = new Align({
      apiKey: process.env.ALIGN_API_KEY!,
      environment: "sandbox",
    });

    const virtualAccount = await align.virtualAccounts.get(
      "va_123e4567-e89b-12d3-a456-426614174000"
    );

    console.log(`Status: ${virtualAccount.status}`);
    console.log(`Currency: ${virtualAccount.source_currency}`);
    console.log(`Deposit to: ${virtualAccount.deposit_instructions.account_number}`);
    ```

  </Tab>
  <Tab title="JavaScript">
    ```javascript
    const virtualAccount = await align.virtualAccounts.get(
      "va_123e4567-e89b-12d3-a456-426614174000"
    );

    console.log("Status:", virtualAccount.status);
    console.log("Account Number:", virtualAccount.deposit_instructions.account_number);
    ```

  </Tab>
</Tabs>

## Error Handling

```typescript
import { AlignError } from "@tolbel/align";

try {
  const account = await align.virtualAccounts.get("invalid-id");
} catch (error) {
  if (error instanceof AlignError && error.statusCode === 404) {
    console.error("Virtual account not found");
  }
}
```
