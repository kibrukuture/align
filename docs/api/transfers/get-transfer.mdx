# Get Transfer

Retrieve detailed information about a specific transfer, including its current status.

## Method Signature

```typescript
align.transfers.get(transferId: string): Promise<Transfer>
```

## Parameters

<ParamField path="transferId" type="string" required>
  The unique transfer identifier
</ParamField>

## Returns

<ResponseField name="transfer_id" type="string">
  Unique transfer identifier
</ResponseField>

<ResponseField name="status" type="string">
  Current status: `pending`, `processing`, `completed`, `failed`, `cancelled`
</ResponseField>

<ResponseField name="type" type="'onramp' | 'offramp'">
  Transfer direction
</ResponseField>

<ResponseField name="source_amount" type="string">
  Amount sent
</ResponseField>

<ResponseField name="destination_amount" type="string">
  Amount received
</ResponseField>

## Examples

<Tabs>
  <Tab title="TypeScript">
    ```typescript
    import Align from "@tolbel/align";

    const align = new Align({
      apiKey: process.env.ALIGN_API_KEY!,
      environment: "sandbox",
    });

    const transfer = await align.transfers.get(
      "tr_123e4567-e89b-12d3-a456-426614174000"
    );

    console.log(`Status: ${transfer.status}`);
    console.log(`Type: ${transfer.type}`);
    console.log(`Amount: ${transfer.source_amount} → ${transfer.destination_amount}`);
    ```

### Poll for Completion

```typescript
async function waitForTransfer(
  transferId: string,
  maxAttempts = 60
): Promise<Transfer> {
  for (let i = 0; i < maxAttempts; i++) {
    const transfer = await align.transfers.get(transferId);

    if (transfer.status === "completed") {
      console.log("✅ Transfer completed!");
      return transfer;
    }

    if (transfer.status === "failed") {
      throw new Error(`Transfer failed: ${transfer.transfer_id}`);
    }

    // Wait 10 seconds
    await new Promise((resolve) => setTimeout(resolve, 10000));
  }

  throw new Error("Transfer timeout");
}
```

<Tip>
  Use [webhooks](/docs/api/webhooks/create) instead of polling for production
  applications.
</Tip>
